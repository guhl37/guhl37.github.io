#include<bits/stdc++.h>
#define int long long
using namespace std;
int s[100005],a[100005];
int f[2][1000005],n,m,ans,now,cnt,tmp;
int flag;
int maxn=-999;
int lowbit(int x) {
	return x&-x;
}
int update(int x,int y) {
	for(; x<=1000000; x+=lowbit(x)) f[y][x]++;
}
int sum(int x,int y) {
	int ret=0;
	for(; x; x-=lowbit(x)) ret+=f[y][x];
	return ret;
}
signed main() {
//	freopen(".in","r",stdin);
//	freopen(".out","w",stdout);
	cin>>n;
	for(int i=1; i<=n; ++i) {
		scanf("%lld",&s[i]);
		s[i]+=s[i-1];
		maxn=max(s[i],maxn);
	}
	for(int i=0; i<=20; ++i) {
		if((1<<i)>maxn) break;
		memset(f,0,sizeof(f));
		flag=0,cnt=0;
		update(1,0);
		for(int j=1; j<=n; ++j) {
			tmp=s[j]&(1<<i);
			if(tmp) now=sum(a[j]+1,0)+sum(1000000,1)-sum(a[j]+1,1);
			else now=sum(a[j]+1,1)+sum(1000000,0)-sum(a[j]+1,0);
			if(now%2) cnt^=1;
			update(a[j]+1,(tmp>0?1:0));
			a[j]|=tmp;
		}
		if(cnt) ans+=(1<<i);
	}
	cout<<ans;
	return 0;
}
